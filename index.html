  name: "drive",
  settings: {
    clientId: "CLIENT_ID",
    redirectURL: "http://localhost:8000/redirector.html",
    ...
  }
};

var gDrive = authz.services.drive;
var pipeline = AeroGear.Pipeline({ authorizer: gDrive });
// Create a pipe

  { name: "files",

  }

    [config setName:@"/team/developers"];
}];

// Read
